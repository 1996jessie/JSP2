drop table empl;
drop sequence e_seq;

create table empl(
	eid number primary key, 
	name varchar2(10) not null,
	dept number not null, 
	salary number 
);

create sequence e_seq
	increment by 1
	start with 1 
	minvalue 1
	maxvalue 10000
	nocache;
